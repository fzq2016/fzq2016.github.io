UIWebView

1 概念性知识
    00 UIWebView是iOS内置的浏览器控件
    01 webView是有缺点的，会导致内存泄露，而且这个问题是它系统本身的问题。
    02 手机上面的safai其实就是用webView来实现的
    03 现在的开发并不完全是原生的开发，而更加倾向于原生+Html5的方式
    04 webView是OC代码和html代码之间进行交互的桥梁
    05 UIWebView不但能加载远程的网页资源，还能加载绝大部分的常见文件,如html、htm、pdf、doc、ppt、txt、mp4..

@interface ViewController ()<UIWebViewDelegate>
@property (weak, nonatomic) IBOutlet UIWebView *webView;

@property (weak, nonatomic) IBOutlet UIBarButtonItem *goBack;
@property (weak, nonatomic) IBOutlet UIBarButtonItem *goforWard;
@end

@implementation ViewController

- (void)viewDidLoad
{
    [super viewDidLoad];
    [self test];
}
- (IBAction)goBackBtnClick:(id)sender
{
    [self.webView goBack];
}
- (IBAction)goForWardBtnClick:(id)sender
{
    [self.webView goForward];
}
- (IBAction)reloadBtnClick:(id)sender
{
    [self.webView reload];
}

-(void)test
{
    //创建请求对象
    NSURL *url = [NSURL URLWithString:@"http://www.baidu.com"];

    NSURLRequest *request = [NSURLRequest requestWithURL:url];

    //传一个请求对象
    [self.webView loadRequest:request];

    //设置代理方法
    self.webView.delegate = self;
}


-(void)webViewDidFinishLoad:(UIWebView *)webView
{
    self.goBack.enabled = self.webView.canGoBack;
    self.goforWard.enabled = self.webView.canGoForward;
}

OC和javascript代码的互调

  • JavaScript是一门脚本语言，简称JS
  • JS的常见作用有
    • 给HTML网页添加动态功能，比如响应用户的各种操作
    • 操纵HTML元素，比如添加、删除、修改网页元素
01 OC调用JS的代码
    NSString *str = [self.webView stringByEvaluatingJavaScriptFromString:@"sum()"];

02 JS怎么调用OC的说明
    新的需求：点击按钮的时候拨打电话
    但是我在点击按钮的时候，用户是不知道的，我们怎么能够知道用户点击了网页上面的一个按钮，只能通过一个技巧，那就是自己搞一个特定的协议头比如说xmg://,当我拦截到你的网络请求的时候，只需要判断一下当前的协议头是不是这个就能判断你现在是否是JS调用。
    OC里面有通过字符串生成SEL类型的方法，所以当拿到数据之后做下面的事情
    1）截取方法的名称
    2）将截取出来的字符串转换为SEL
    3）利用performSelect方法来调用SEL

03 涉及到的相关方法
    [@"abc" hasPrefix:@"A"] //判断字符串是否以一个固定的字符开头，这里为A

-(BOOL)webView:(UIWebView *)webView shouldStartLoadWithRequest:(NSURLRequest *)request navigationType:(UIWebViewNavigationType)navigationType
{
    NSLog(@"%@",request.URL.absoluteString);
    NSString *urlStr = request.URL.absoluteString;
    NSString *format = @"xiaomage://";

    if ([urlStr hasPrefix:format]) {
        //来到这里,说明需要调用OC的方法
        //截取方法的名称和参数
        NSString *strM = [urlStr substringFromIndex:format.length];
        NSArray *subStrs = [strM componentsSeparatedByString:@"?"];
        NSLog(@"%@",subStrs);

        //获得要调用的方法
        //第一个参数:要替换的是谁  _
        //第二个参数:要用什么来替换:
        NSString *methedstr =  [[subStrs firstObject] stringByReplacingOccurrencesOfString:@"_" withString:@":"];
        NSLog(@"%@",methedstr);

        //把字符串转换为SEL
        SEL selector = NSSelectorFromString(methedstr);

        //处理参数的问题
        NSString *paramStr = [subStrs lastObject];
        NSArray *subParams = [paramStr componentsSeparatedByString:@"&"];
        NSLog(@"%@",subParams);

        [self performSelector:selector withObjects:subParams];
        return NO;
    }

    return YES;
}
